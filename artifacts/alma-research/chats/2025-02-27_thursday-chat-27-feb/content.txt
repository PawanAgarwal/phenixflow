Here's a trading view script that plots the levels from almas post today on the SPX chart. I find it helpful to use this along with her spx Standard deviation levels.  //@version=5
indicator("GEX and Vanna Levels", overlay=true)

// Ensure the script only runs on SPX
isSPX = syminfo.tickerid == "SP:SPX" or syminfo.tickerid == "SPX" or syminfo.tickerid == "SPX500USD"

// User settings to toggle individual elements (levels from narrative)
showMainPivot          = input(true, title="Show Main Pivot 6015.90")
showUpsideLongGamma    = input(true, title="Show Upside Long Gamma 6068.72")
showZeroVannaLow       = input(true, title="Show Local Zero Vanna 5911.65")
showZeroVannaHigh      = input(true, title="Show Local Zero Vanna 5982")
showGammaResistance    = input(true, title="Show Gamma Resistance 6000")
showNetZeroSpeed       = input(true, title="Show Net Zero Speed 5856.06")
showSupportiveCharm    = input(true, title="Show Supportive Charm 5926")

// Define key levels from the narrative
main_pivot          = 6015.90   // Main pivot from momentum perspective; if bullish, spot should take this out for upside gamma support.
upside_long_gamma   = 6068.72   // Level where upside long gamma starts to slow momentum and temper volatility.
zero_vanna_low      = 5911.65   // Local zero vanna (lower end of oscillation in the 5911.65-5919 range); requires lower local vol to break.
zero_vanna_high     = 5982.00   // Second local zero vanna line; requires lower local vol to break.
gamma_resistance    = 6000.00   // Strong gamma resistance due to high local DEX and potential pin.
net_zero_speed      = 5856.06   // Net zero speed line marking the MM net short skew/long speed environment.
supportive_charm    = 5926.00   // Level above which supportive charm is present, even as overall gamma exposure decreases.

// Colors and label settings
mainPivot_color         = color.green
upsideGamma_color       = color.blue
zeroVanna_color         = color.orange
gammaResistance_color   = color.red
netZeroSpeed_color      = color.gray
supportiveCharm_color   = color.purple
transparent_bg          = color.new(color.white, 10)
label_size              = size.normal
label_style             = label.style_none

// Plot horizontal dashed lines for each level
hline_main_pivot        = hline(showMainPivot ? main_pivot : na, "Main Pivot 6015.90", showMainPivot ? mainPivot_color : na, hline.style_dashed, 2)
hline_upside_gamma      = hline(showUpsideLongGamma ? upside_long_gamma : na, "Upside Long Gamma 6068.72", showUpsideLongGamma ? upsideGamma_color : na, hline.style_dashed, 2)
hline_zero_vanna_low    = hline(showZeroVannaLow ? zero_vanna_low : na, "Local Zero Vanna 5911.65", showZeroVannaLow ? zeroVanna_color : na, hline.style_dashed, 2)
hline_zero_vanna_high   = hline(showZeroVannaHigh ? zero_vanna_high : na, "Local Zero Vanna 5982", showZeroVannaHigh ? zeroVanna_color : na, hline.style_dashed, 2)
hline_gamma_resistance  = hline(showGammaResistance ? gamma_resistance : na, "Gamma Resistance 6000", showGammaResistance ? gammaResistance_color : na, hline.style_dashed, 2)
hline_net_zero_speed    = hline(showNetZeroSpeed ? net_zero_speed : na, "Net Zero Speed 5856.06", showNetZeroSpeed ? netZeroSpeed_color : na, hline.style_dashed, 2)
hline_supportive_charm  = hline(showSupportiveCharm ? supportive_charm : na, "Supportive Charm 5926", showSupportiveCharm ? supportiveCharm_color : na, hline.style_dashed, 2)

// Detect the first and last bar of the current trading day
var int first_bar_of_day = na
var int last_bar_of_day  = na
first_bar_of_day := ta.change(dayofweek) ? bar_index : nz(first_bar_of_day)
last_bar_of_day  := first_bar_of_day + 390  // Approximate full trading session

// Create labels for each level on the current trading day
var label mainPivotLabel         = na
var label upsideGammaLabel       = na
var label zeroVannaLowLabel      = na
var label zeroVannaHighLabel     = na
var label gammaResistanceLabel   = na
var label netZeroSpeedLabel      = na
var label supportiveCharmLabel   = na

if isSPX and bar_index >= first_bar_of_day and bar_index <= last_bar_of_day
    if showMainPivot
        mainPivotLabel := label.new(first_bar_of_day, main_pivot, "Main Pivot 6015.90", xloc=xloc.bar_index, color=transparent_bg, size=label_size, style=label_style, textcolor=mainPivot_color, textalign=text.align_center)
    if showUpsideLongGamma
        upsideGammaLabel := label.new(first_bar_of_day, upside_long_gamma, "Upside Long Gamma 6068.72", xloc=xloc.bar_index, color=transparent_bg, size=label_size, style=label_style, textcolor=upsideGamma_color, textalign=text.align_center)
    if showZeroVannaLow
        zeroVannaLowLabel := label.new(first_bar_of_day, zero_vanna_low, "Local Zero Vanna 5911.65", xloc=xloc.bar_index, color=transparent_bg, size=label_size, style=label_style, textcolor=zeroVanna_color, textalign=text.align_center)
    if showZeroVannaHigh
        zeroVannaHighLabel := label.new(first_bar_of_day, zero_vanna_high, "Local Zero Vanna 5982", xloc=xloc.bar_index, color=transparent_bg, size=label_size, style=label_style, textcolor=zeroVanna_color, textalign=text.align_center)
    if showGammaResistance
        gammaResistanceLabel := label.new(first_bar_of_day, gamma_resistance, "Gamma Resistance 6000", xloc=xloc.bar_index, color=transparent_bg, size=label_size, style=label_style, textcolor=gammaResistance_color, textalign=text.align_center)
    if showNetZeroSpeed
        netZeroSpeedLabel := label.new(first_bar_of_day, net_zero_speed, "Net Zero Speed 5856.06", xloc=xloc.bar_index, color=transparent_bg, size=label_size, style=label_style, textcolor=netZeroSpeed_color, textalign=text.align_center)
    if showSupportiveCharm
        supportiveCharmLabel := label.new(first_bar_of_day, supportive_charm, "Supportive Charm 5926", xloc=xloc.bar_index, color=transparent_bg, size=label_size, style=label_style, textcolor=supportiveCharm_color, textalign=text.align_center)
