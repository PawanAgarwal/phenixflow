// Ensure the script only runs on SPX
isSPX = syminfo.tickerid == "SP:SPX" or syminfo.tickerid == "SPX" or syminfo.tickerid == "SPX500USD"

// User settings to toggle individual elements
showButterflySpread  = input(true, title="Show Butterfly Spread 5885/5965/6050")
showCentroid         = input(true, title="Show Centroid 5965.94")
showDownsideGamma    = input(true, title="Show Downside Gamma 5925.33")
showZeroVanna        = input(true, title="Show Zero Vanna 5988.5")

// Define key levels from the narrative
butterfly_lower      = 5885.0
butterfly_centroid   = 5965.94
butterfly_upper      = 6050.0

downside_gamma       = 5925.33
zero_vanna           = 5988.5

// Colors and label settings
centroid_color      = color.green
downside_color      = color.red
zero_vanna_color    = color.orange
transparent_bg      = color.new(color.gray, 10)
label_size          = size.normal
label_style         = label.style_none

// Plot horizontal dashed lines for each level if toggled
hline_centroid = hline(showCentroid ? butterfly_centroid : na, "Centroid 5965.94", showCentroid ? centroid_color : na, hline.style_dashed, 2)
hline_downside = hline(showDownsideGamma ? downside_gamma : na, "Downside Gamma 5925.33", showDownsideGamma ? downside_color : na, hline.style_dashed, 2)
hline_zero_vanna = hline(showZeroVanna ? zero_vanna : na, "Zero Vanna 5988.5", showZeroVanna ? zero_vanna_color : na, hline.style_dashed, 2)

// Detect the first and last bar of the current trading day
var int first_bar_of_day = na
var int last_bar_of_day  = na
first_bar_of_day := ta.change(dayofweek) ? bar_index : nz(first_bar_of_day)
last_bar_of_day  := first_bar_of_day + 390  // Approximate full trading session

// Draw butterfly spread horizontal box for the current trading day if toggled
var box butterflyBox = na
if isSPX and bar_index == first_bar_of_day and showButterflySpread
    butterflyBox := box.new(first_bar_of_day, butterfly_upper, last_bar_of_day, butterfly_lower, xloc=xloc.bar_index, border_color=color.new(color.gray, 0), border_width=1, bgcolor=color.new(color.gray, 90))

// Create labels for each level if toggled, only on the current trading day
var label centroidLabel = na
var label downsideGammaLabel = na
var label zeroVannaLabel = na

if isSPX and bar_index >= first_bar_of_day and bar_index <= last_bar_of_day
    if showCentroid
        centroidLabel := label.new(first_bar_of_day, butterfly_centroid, "Centroid 5965.94", xloc=xloc.bar_index, color=transparent_bg, size=label_size, style=label_style, textcolor=centroid_color, textalign=text.align_center)
    if showDownsideGamma
        downsideGammaLabel := label.new(first_bar_of_day, downside_gamma, "Downside Gamma 5925.33", xloc=xloc.bar_index, color=transparent_bg, size=label_size, style=label_style, textcolor=downside_color, textalign=text.align_center)
    if showZeroVanna
        zeroVannaLabel := label.new(first_bar_of_day, zero_vanna, "Zero Vanna 5988.5", xloc=xloc.bar_index, color=transparent_bg, size=label_size, style=label_style, textcolor=zero_vanna_color, textalign=text.align_center)

// Alert for a potential breakthrough of the Zero Vanna level
alertcondition(close > zero_vanna, title="Zero Vanna Breakthrough", message="Price breaking above Zero Vanna 5988.5 may indicate volatility contraction")
