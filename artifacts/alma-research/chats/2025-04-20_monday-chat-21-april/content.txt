# Input Variables
input src = close;
input mult = 1.5; # 1.5 Sigma
input mult2 = 2.5; # 2.5 Sigma
input mult3 = 0.674; # Epsilon
input window = 20;
input trading_periods = 252;

# Custom VWMA Function
def sumPriceVol = fold i = 0 to window
                  with p = 0
                  do p + GetValue(src, i) * GetValue(volume, i);
def sumVol = fold j = 0 to window
             with v = 0
             do v + GetValue(volume, j);
def VWMA = if window > 0 and sumVol != 0 then sumPriceVol / sumVol else Double.NaN;

# Daily Realized Volatility Calculation
def log_hoD = Log(high("period" = "DAY") / open("period" = "DAY"));
def log_loD = Log(low("period" = "DAY") / open("period" = "DAY"));
def log_coD = Log(close("period" = "DAY") / open("period" = "DAY"));

def log_ocD = Log(open("period" = "DAY") / close("period" = "DAY")[1]);
def log_oc_sqD = log_ocD * log_ocD;

def log_ccD = Log(close("period" = "DAY") / close("period" = "DAY")[1]);
def log_cc_sqD = log_ccD * log_ccD;

def rsD = log_hoD * (log_hoD - log_coD) + log_loD * (log_loD - log_coD);

def close_volD = Sum(log_cc_sqD, window) / (window - 1);
def open_volD = Sum(log_oc_sqD, window) / (window - 1);
def window_rsD = Sum(rsD, window) / (window - 1);
def kD = 0.34 / (1.34 + (window + 1.0) / (window - 1.0));
def rvD = Sqrt((open_volD + kD * close_volD + (1 - kD) * window_rsD) * trading_periods);

# Standard Deviation Bands
def basis = VWMA;
def dev = mult * StDev(src, window);
def dev2 = mult2 * StDev(src, window);
def dev3 = mult3 * StDev(src, window);

def upper = basis + dev;
def lower = basis - dev;
def upper2 = basis + dev2;
def lower2 = basis - dev2;
def upper3 = basis + dev3;
def lower3 = basis - dev3;

# Plotting
plot Basi = basis;
Basi.SetDefaultColor(Color.ORANGE);
Basi.SetLineWeight(2);

plot Uper = upper;
plot Lowr = lower;
Uper.SetDefaultColor(Color.MAGENTA);
Lowr.SetDefaultColor(Color.MAGENTA);

plot UPER2 = upper2;
plot Lowr2 = lower2;
UPER2.SetDefaultColor(Color.VIOLET);
Lowr2.SetDefaultColor(Color.VIOLET);

plot Uper3 = upper3;
plot Lowr3 = lower3;
Uper3.SetDefaultColor(Color.GRAY);
Lowr3.SetDefaultColor(Color.GRAY);
Uper3.SetStyle(Curve.SHORT_DASH);
Lowr3.SetStyle(Curve.SHORT_DASH);

# Input Variables
input src = close;
input mult = 1.5; # 1.5 Sigma
input mult2 = 2.5; # 2.5 Sigma
input mult3 = 0.674; # Epsilon
input window = 20;
input trading_periods = 252;
input showLabels = yes;

# Custom VWMA Function
def sumPriceVol = fold i = 0 to window
                  with p = 0
                  do p + GetValue(src, i) * GetValue(volume, i);
def sumVol = fold j = 0 to window
             with v = 0
             do v + GetValue(volume, j);
def VWMA = if window > 0 and sumVol != 0 then sumPriceVol / sumVol else Double.NaN;

# Daily Realized Volatility Calculation
def log_hoD = Log(high("period" = "DAY") / open("period" = "DAY"));
def log_loD = Log(low("period" = "DAY") / open("period" = "DAY"));
def log_coD = Log(close("period" = "DAY") / open("period" = "DAY"));

def log_ocD = Log(open("period" = "DAY") / close("period" = "DAY")[1]);
def log_oc_sqD = log_ocD * log_ocD;

def log_ccD = Log(close("period" = "DAY") / close("period" = "DAY")[1]);
def log_cc_sqD = log_ccD * log_ccD;

def rsD = log_hoD * (log_hoD - log_coD) + log_loD * (log_loD - log_coD);

def close_volD = Sum(log_cc_sqD, window) / (window - 1);
def open_volD = Sum(log_oc_sqD, window) / (window - 1);
def window_rsD = Sum(rsD, window) / (window - 1);
def kD = 0.34 / (1.34 + (window + 1.0) / (window - 1.0));
def rvD = Sqrt((open_volD + kD * close_volD + (1 - kD) * window_rsD) * trading_periods);

# Daily RV Rate of Change
def dailyRV_yester = rvD[1];
def dailyRoc = if dailyRV_yester != 0 then (rvD - dailyRV_yester) / dailyRV_yester * 100.0 else 0;

# Standard Deviation Bands
def basis = VWMA;
def roc = Round((close - close[1]) / close[1] * 100, 2);
def rv_rounded = Round(rvD, 4);
def roc2 = Round(dailyRoc, 2);
def dev = mult * StDev(src, window);
def dev2 = mult2 * StDev(src, window);
def dev3 = mult3 * StDev(src, window);

def upper = basis + dev;
def lower = basis - dev;
def upper2 = basis + dev2;
def lower2 = basis - dev2;
def upper3 = basis + dev3;
def lower3 = basis - dev3;

# Plotting
plot Basi = basis;
Basi.SetDefaultColor(Color.ORANGE);
Basi.SetLineWeight(2);

plot Uper = upper;
plot Lowr = lower;
Uper.SetDefaultColor(Color.MAGENTA);
Lowr.SetDefaultColor(Color.MAGENTA);

plot UPER2 = upper2;
plot Lowr2 = lower2;
UPER2.SetDefaultColor(Color.VIOLET);
Lowr2.SetDefaultColor(Color.VIOLET);

plot Uper3 = upper3;
plot Lowr3 = lower3;
Uper3.SetDefaultColor(Color.GRAY);
Lowr3.SetDefaultColor(Color.GRAY);
Uper3.SetStyle(Curve.SHORT_DASH);
Lowr3.SetStyle(Curve.SHORT_DASH);

# Fill Between Bands

#AddCloud(Uper3, Lowr3, Color.DARK_GRAY, Color.DARK_GRAY);

# Labels (Replacement for Table)
addlabel(showLabels, "Change %: " + roc + "%", Color.LIGHT_GRAY);
addlabel(showLabels, "20D RV: " + (rv_rounded * 100) + "%", Color.LIGHT_GRAY);
addlabel(showLabels, "RV Change %: " + roc2 + "%", Color.LIGHT_GRAY);
addlabel(showLabels, "1 SD Upper: " + uper, Color.LIGHT_GRAY);

addlabel(showLabels, "2 SD Lower: " + lowr2, Color.LIGHT_GRAY);
