import numpy as np
import math

def standard_deviation(price_data, window=20, trading_periods=252, clean=True):
    log_return = (price_data["Close"] / price_data["Close"].shift(1)).apply(np.log)
    squared_log_return = log_return ** 2
    rolling_squared_log_return = squared_log_return.rolling(window=window, center=False).sum() / (window - 1)
    rolling_mean_log_return = log_return.rolling(window=window, center=False).mean()
    squared_rolling_mean = (rolling_mean_log_return ** 2) / (window * (window - 1))
    adjusted_variance = rolling_squared_log_return - squared_rolling_mean / window
    adjusted_std_dev = adjusted_variance.apply(np.sqrt) * math.sqrt(trading_periods)

    if clean:
        return adjusted_std_dev.dropna()
    else:
        return adjusted_std_dev

rolling_std_dev = standard_deviation(price_data)
print(standard_deviation(price_data))

plt.style.use('dark_background')

plt.title(f'Rolling SD of Logreturns for {symbol}')
plt.xlabel('Date')
plt.ylabel('Annualized SD')
plt.legend()
plt.show()

def parkinson(price_data, window=20, trading_periods=252, clean=True):
    rs = (1.0 / (4.0 * math.log(2.0))) * (np.log(price_data["High"] / price_data["Low"])) ** 2.0

    def f(v):
        return (trading_periods * v.mean()) ** 0.5

    result = rs.rolling(window=window, center=False).apply(func=f, raw=True)

    if clean:
        return result.dropna()
    else:
        return result

p_volatility = parkinson(price_data)

def garman_klass(price_data, window=20, trading_periods=252, clean=True):
    if not all(col in price_data.columns for col in ['High', 'Low', 'Close', 'Open']):
        raise ValueError("price_data must contain 'High', 'Low', 'Close', and 'Open' columns.")

    log_high_low = np.log(price_data['High'] / price_data['Low'])
    log_close_open = np.log(price_data['Close'] / price_data['Open'])

    gk_volatility = (
        (0.5 * log_high_low ** 2) -
        ((2 * np.log(2) - 1) * log_close_open ** 2)
    )

    gk_volatility = np.sqrt(trading_periods * gk_volatility.rolling(window=window).mean())

    if clean:
        gk_volatility = gk_volatility.dropna()

    return gk_volatility

def rogers_satchell(price_data, window=20, trading_periods=252, clean=True):
    if not all(col in price_data.columns for col in ['High', 'Low', 'Close', 'Open']):
        raise ValueError("price_data must contain 'High', 'Low', 'Close', and 'Open' columns.")

    log_high_open = np.log(price_data['High'] / price_data['Open'])
    log_low_open = np.log(price_data['Low'] / price_data['Open'])
    log_close_open = np.log(price_data['Close'] / price_data['Open'])

    rs_volatility = (
        (log_high_open * (log_high_open - log_close_open)) +
        (log_low_open * (log_low_open - log_close_open))
    )

    rs_volatility = np.sqrt(trading_periods * rs_volatility.rolling(window=window).mean())

    if clean:
        rs_volatility = rs_volatility.dropna()

    return rs_volatility

def yang_zhang(price_data, window=20, trading_periods=252, clean=True):

    log_ho = (price_data['High'] / price_data['Open']).apply(np.log)
    log_lo = (price_data['Low'] / price_data['Open']).apply(np.log)
    log_co = (price_data['Close'] / price_data['Open']).apply(np.log)

    log_oc = (price_data['Open'] / price_data['Close'].shift(1)).apply(np.log)
    log_oc_sq = log_oc**2

    log_cc = (price_data['Close'] / price_data['Close'].shift(1)).apply(np.log)
    log_cc_sq = log_cc**2

    rs = log_ho * (log_ho - log_co) + log_lo * (log_lo - log_co)

    close_vol = log_cc_sq.rolling(
        window=window,
        center=False
    ).sum() * (1.0 / (window - 1.0))
    open_vol = log_oc_sq.rolling(
        window=window,
        center=False
    ).sum() * (1.0 / (window - 1.0))
    window_rs = rs.rolling(
        window=window,
        center=False
    ).sum() * (1.0 / (window - 1.0))

    k = 0.34 / (1.34 + (window + 1) / (window - 1))
    result = (open_vol + k * close_vol + (1 - k) * window_rs).apply(np.sqrt) * math.sqrt(trading_periods)

    if clean:
        return result.dropna()
    else:
        return result
