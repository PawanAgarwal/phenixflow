TxtToArray(string origin, string divider) =>

// Updated parsing logic supporting old and new formats
ParseSymbolThresholds(string inputText) =>
    symbolThresholds = array.new_string()
    if str.length(inputText) > 0
        lines = TxtToArray(inputText, '\n')
        currentSymbol = ''
        for i = 0 to array.size(lines) - 1
            line = str.trim(array.get(lines, i))

            // New format: === SYMBOL ===
            if str.startswith(line, '===') and str.endswith(line, '===') and str.length(line) > 7
                words = TxtToArray(line," ")
                symbolName = array.get(words,1)
                currentSymbol := symbolName
            // Old format: line without commas and not empty
            else if not str.contains(line, ',') and str.length(line) > 0
                currentSymbol := str.trim(line)
            // If this line is data and currentSymbol is set
            else if str.contains(line, ',') and str.length(currentSymbol) > 0
                array.push(symbolThresholds, currentSymbol)
                array.push(symbolThresholds, line)
                currentSymbol := ''
    symbolThresholds

// Normalize and match against current symbol
GetCurrentSymbolThresholds(array<string> symbolThresholds, string currentSymbol) =>
    result = ''
    normalizedCurrent = str.upper(currentSymbol)
    for i = 0 to array.size(symbolThresholds) - 1 by 2
        if i + 1 < array.size(symbolThresholds)
            symbol = array.get(symbolThresholds, i)
            thresholds = array.get(symbolThresholds, i + 1)

            normalizedInput = str.upper(symbol)
            normalizedInput := str.replace(normalizedInput, '=F', '1!')
            normalizedInput := normalizedInput == 'ES' ? 'ES1!' : normalizedInput

            if normalizedInput == normalizedCurrent
                result := thresholds
                break
    result
