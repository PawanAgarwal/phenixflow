//@version=5
indicator("Rolling VIX Points per 1% SPX Move", shorttitle="VIX Pts/SPX %", overlay=false)

// Inputs
lookback = input.int(20, title="Lookback Period", minval=1)

// Get SPX and VIX data
spx_close = request.security("SPX", timeframe.period, close)
vix_close = request.security("VIX", timeframe.period, close)

// Calculate SPX % returns and VIX point changes
spx_returns = math.log(spx_close / nz(spx_close[1])) * 100  // SPX % change
vix_points = vix_close - nz(vix_close[1])                    // VIX point change

// Calculate rolling covariance and SPX variance
covariance = math.sum(spx_returns * vix_points, lookback) / (lookback - 1) - (math.sum(spx_returns, lookback) * math.sum(vix_points, lookback)) / (lookback * (lookback - 1))
spx_variance = math.sum(spx_returns * spx_returns, lookback) / (lookback - 1) - math.pow(math.sum(spx_returns, lookback), 2) / (lookback * (lookback - 1))

// Calculate "beta" as VIX points per 1% SPX move
beta = covariance / spx_variance

// Plot the result
plot(beta, title="VIX Points per 1% SPX Move", color=color.blue, linewidth=2)
